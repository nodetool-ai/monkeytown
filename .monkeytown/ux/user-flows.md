### Memory Echo Implementation

```typescript
interface MemoryEcho {
  type: 'time' | 'specific_move' | 'relationship' | 'continuous';
  strength: number; // 1-5
  message: string;
  visual: 'glow' | 'sparkle' | 'pulse' | 'aurora';
}

function generateMemoryEcho(player: Player, agent: Agent): MemoryEcho {
  // Check time since last visit
  const daysSince = daysSince(player.lastVisit);
  if (daysSince > 2) {
    return {
      type: 'time',
      strength: 3,
      message: `It's been ${daysSince} days since your last visit.`,
      visual: 'aurora'
    };
  }

  // Check for specific memorable moves
  const memorableMoves = player.history
    .filter(h => h.agentId === agent.id && h.memorable);
  if (memorableMoves.length > 0) {
    const lastMove = memorableMoves[memorableMoves.length - 1];
    return {
      type: 'specific_move',
      strength: 5,
      message: `I've been thinking about your ${lastMove.notation} move.`,
      visual: 'sparkle'
    };
  }

  // Check streak
  const streak = calculateStreak(player, agent);
  if (streak > 3) {
    return {
      type: 'relationship',
      strength: 4,
      message: `${streak} games together. Your strategies are improving.`,
      visual: 'glow'
    };
  }

  return null;
}
```
